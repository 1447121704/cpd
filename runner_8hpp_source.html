115  typename Transform::Result run(const Matrix& fixed, const Matrix& moving) {
116  if (m_logger) {
117  m_logger->info("Number of points in fixed matrix: {}",
118  fixed.rows());
119  m_logger->info("Number of points in moving matrix: {}",
120  moving.rows());
121  }
122  const Matrix* fixed_ptr(&fixed);
123  const Matrix* moving_ptr(&moving);
124  Normalization normalization;
125  const auto tic = std::chrono::high_resolution_clock::now();
126  if (m_normalize) {
127  normalization = cpd::normalize(fixed, moving);
128  fixed_ptr = &normalization.fixed;
129  moving_ptr = &normalization.moving;
130  }
131  m_transform.init(*fixed_ptr, *moving_ptr);
132  size_t iter = 0;
133  double ntol = m_tolerance + 10.0;
134  double l = 0.0;
135  typename Transform::Result result;
136  result.points = *moving_ptr;
137  if (m_sigma2 == 0.0) {
138  result.sigma2 = default_sigma2(*fixed_ptr, *moving_ptr);
139  if (m_logger) {
140  m_logger->info("Initializing sigma2 to {}", result.sigma2);
141  }
142  } else {
143  result.sigma2 = m_sigma2;
144  if (m_logger) {
145  m_logger->info("sigma2 previously set to {}", result.sigma2);
146  }
147  }
148  while (iter < m_max_iterations && ntol > m_tolerance &&
149  result.sigma2 > 10 * std::numeric_limits<double>::epsilon()) {
150  auto probabilities = m_probability_computer.compute(
151  *fixed_ptr, result.points, result.sigma2, m_outliers);
152  m_transform.modify_probabilities(probabilities);
153  ntol = std::abs((probabilities.l - l) / probabilities.l);
154  if (m_logger) {
155  m_logger->info("iter={}, dL={:.8f}, sigma2={:.8f}", iter, ntol,
156  result.sigma2);
157  }
158  l = probabilities.l;
159  result = m_transform.compute(*fixed_ptr, *moving_ptr, probabilities,
160  result.sigma2);
161  ++iter;
162  }
163  if (m_normalize) {
164  m_transform.denormalize(normalization, result);
165  }
166  if (m_correspondence) {
167  auto probabilities = DirectComparer().compute(fixed, result.points,
168  m_sigma2, m_outliers);
169  result.correspondence = probabilities.correspondence;
170  }
171  const auto toc = std::chrono::high_resolution_clock::now();
172  result.runtime =
173  std::chrono::duration_cast<std::chrono::microseconds>(toc - tic);
174  result.iterations = iter;
175  return result;
176  }
