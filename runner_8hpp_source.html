131  typename Transform::Result run(const Matrix& fixed, const Matrix& moving) {
132  if (m_logger) {
133  m_logger->info("Number of points in fixed matrix: {}",
134  fixed.rows());
135  m_logger->info("Number of points in moving matrix: {}",
136  moving.rows());
137  }
138  const Matrix* fixed_ptr(&fixed);
139  const Matrix* moving_ptr(&moving);
140  Normalization normalization;
141  const auto tic = std::chrono::high_resolution_clock::now();
142  if (m_normalize) {
143  normalization = cpd::normalize(fixed, moving);
144  fixed_ptr = &normalization.fixed;
145  moving_ptr = &normalization.moving;
146  }
147  m_transform.init(*fixed_ptr, *moving_ptr);
148  size_t iter = 0;
149  double ntol = m_tolerance + 10.0;
150  double l = 0.0;
151  typename Transform::Result result;
152  result.points = *moving_ptr;
153  if (m_sigma2 == 0.0) {
154  result.sigma2 = default_sigma2(*fixed_ptr, *moving_ptr);
155  if (m_logger) {
156  m_logger->info("Initializing sigma2 to {}", result.sigma2);
157  }
158  } else {
159  result.sigma2 = m_sigma2;
160  if (m_logger) {
161  m_logger->info("sigma2 previously set to {}", result.sigma2);
162  }
163  }
164  while (iter < m_max_iterations && ntol > m_tolerance &&
165  result.sigma2 > 10 * std::numeric_limits<double>::epsilon()) {
166  auto probabilities = m_comparer->compute(*fixed_ptr, result.points,
167  result.sigma2, m_outliers);
168  m_transform.modify_probabilities(probabilities);
169  ntol = std::abs((probabilities.l - l) / probabilities.l);
170  if (m_logger) {
171  m_logger->info("iter={}, dL={:.8f}, sigma2={:.8f}", iter, ntol,
172  result.sigma2);
173  }
174  l = probabilities.l;
175  result = m_transform.compute(*fixed_ptr, *moving_ptr, probabilities,
176  result.sigma2);
177  ++iter;
178  }
179  if (m_normalize) {
180  m_transform.denormalize(normalization, result);
181  }
182  if (m_correspondence) {
183  auto probabilities = DirectComparer().compute(fixed, result.points,
184  m_sigma2, m_outliers);
185  result.correspondence = probabilities.correspondence;
186  }
187  const auto toc = std::chrono::high_resolution_clock::now();
188  result.runtime =
189  std::chrono::duration_cast<std::chrono::microseconds>(toc - tic);
190  result.iterations = iter;
191  return result;
192  }
