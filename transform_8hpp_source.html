128  Result run(Matrix fixed, Matrix moving) {
129  auto tic = std::chrono::high_resolution_clock::now();
130  Normalization normalization(fixed, moving);
131  if (m_normalize) {
132  fixed = normalization.fixed;
133  moving = normalization.moving;
134  }
135 
136  this->init(fixed, moving);
137 
138  Result result;
139  result.points = moving;
140  if (m_sigma2 == 0.0) {
141  result.sigma2 = cpd::default_sigma2(fixed, moving);
142  } else if (m_normalize) {
143  result.sigma2 = m_sigma2 / normalization.scale;
144  } else {
145  result.sigma2 = m_sigma2;
146  }
147 
148  size_t iter = 0;
149  double ntol = m_tolerance + 10.0;
150  double l = 0.;
151  while (iter < m_max_iterations && ntol > m_tolerance &&
152  result.sigma2 > 10 * std::numeric_limits<double>::epsilon()) {
153  Probabilities probabilities = m_gauss_transform->compute(
154  fixed, result.points, result.sigma2, m_outliers);
155  this->modify_probabilities(probabilities);
156 
157  ntol = std::abs((probabilities.l - l) / probabilities.l);
158  l = probabilities.l;
159 
160  result =
161  this->compute_one(fixed, moving, probabilities, result.sigma2);
162  ++iter;
163  }
164 
165  if (m_normalize) {
166  result.denormalize(normalization);
167  }
168  if (m_correspondence) {
169  Probabilities probabilities = m_gauss_transform->compute(
170  fixed, result.points, m_sigma2, m_outliers);
171  result.correspondence = probabilities.correspondence;
172  assert(result.correspondence.rows() > 0);
173  }
174  auto toc = std::chrono::high_resolution_clock::now();
175  result.runtime =
176  std::chrono::duration_cast<std::chrono::microseconds>(toc - tic);
177  result.iterations = iter;
178  return result;
179  }
